def pca_kotov(x, r=2):
    
    z = x - x.mean(axis=0)
    
    zfeatures=[z[:, i] for i in range(z.shape[1])]
    
    k = np.cov(zfeatures)
    
    l,u = np.linalg.eig(k)
    
    uv = np.array([u[:,i] for i in range(r)]).transpose()
    
    fr=l/l.sum()
    
    A = z.dot(uv)  
    
    return A, fr
